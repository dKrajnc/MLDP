69MCFolds getFolds( QString& aSettingsPath, const lpmldata::DataPackage& aDataPackage )
70{
71 QSettings settings( aSettingsPath, QSettings::IniFormat );
72
73 //Check parameter validity
74 bool isSplitPercentage;
75 auto splitPercentage = std::abs( settings.value( "CentralAi/splitPercentage" ).toDouble( &isSplitPercentage ) );
76 if ( !isSplitPercentage )
77 {
78 qDebug() << "CentralAi - Error: Invalid parameter splitPercentage";
79 isSplitPercentage = false;
80 }
81
82 bool isFoldCount;
83 auto foldCount = std::abs( settings.value( "CentralAi/foldCount" ).toInt( &isFoldCount ) );
84 if ( !isFoldCount )
85 {
86 qDebug() << "CentralAi - Error: Invalid parameter foldCount";
87 isFoldCount = false;
88 }
89
90
91 //Generate folds
92 auto validationSize = aDataPackage.sampleCountOfPercentage( splitPercentage );
93
94 dkeval::PatientFoldGenerator foldGenerator( aDataPackage, validationSize );
95 foldGenerator.generate( foldCount );
96 qDebug() << "Number of generated folds: " << foldCount;
97
98
99 //Create Monte Carlo cross-validation folds list
100 MCFolds folds;
101
102 for ( int i = 0; i < foldCount; ++i )
103 {
104 auto fold = foldGenerator.fold( i );
105
106 folds.push_back( fold );
107 }
108
109 return folds;
110}

142void singleCenterAnalysis( const QString& aGlobalSettingsPath, const QString& aDataPath )
143{
144 QDir dir( aDataPath );
145 QString settingsPath = aGlobalSettingsPath + "Settings.ini";
146 QString pluginSettingsPath = aGlobalSettingsPath + "pluginSettings.ini";
147 int foldCounter = 0;
148
149 //Load feature and label database from .csv file
150 lpmldata::TabularData FDB;
151 lpmldata::TabularData LDB;
152 lpmlfio::TabularDataFileIo loader;
153 loader.load( aDataPath + "FDB.csv", FDB );
154 loader.load( aDataPath + "LDB.csv", LDB );
155
156 //Check FDB+LDB and generate folds
157 auto optimizedData = optimizeData( FDB, LDB );
158 auto folds = getFolds( settingsPath, optimizedData );
159
160 //Store TP, TN, FP, FN across all folds
161 QMap< QString, double > totalConfusionMatrixValues;
162
163
164 //Run paprallel analysis - singlecenter data
165#pragma omp parallel for schedule( static )
166 for ( int i = 0; i < folds.size(); ++i )
167 {
168 QSettings settings( settingsPath, QSettings::IniFormat );
169 QSettings pluginSettings( pluginSettingsPath, QSettings::IniFormat );
170
171 auto fold = folds.at( i );
172 auto trainingDataPair = fold.first;
173 auto validationDataPair = fold.second;
174
175 dkeval::CentralAi ai( &settings, &pluginSettings, *trainingDataPair, *validationDataPair );
176 ai.setFoldId( i + 1 );
177 ai.execute();
178 qInfo() << "Fold:" << i + 1 << "analysis in progress...";
179
180#pragma omp critical
181 {
182 auto confusionValues = ai.confusionMatrixValues();
183 for ( auto key : confusionValues.keys() )
184 {
185 totalConfusionMatrixValues[ key ] += confusionValues.value( key );
186 }
187
188 foldCounter++;
189 }
190
191
192 //Store folds
193 auto foldPath = aDataPath + "/Folds/Fold-" + QString::number( i + 1 );
194 dir.mkpath( foldPath );
195 ai.saveFoldAt( foldPath );
196
197
198 //Store predctionSummary.csv, performance.csv and pipeline_info.txt
199 foldPath = aDataPath + "/Summary/fold_information/Fold-" + QString::number( i + 1 );
200 dir.mkpath( foldPath );
201
202 ai.savePipelineInfo( foldPath, "/pipeline_info.txt" );
203 ai.savePerformanceInfo( foldPath, "/performance_info.csv" );
204 qInfo() << "Fold:" << i + 1 << "analysis finished and saved!";
205
206
207 //Remove temp files
208 QFile pipelineSettingsFile( aGlobalSettingsPath + QString::number( i + 1 ) + "pipelineSettings.ini" );
209 pipelineSettingsFile.remove();
210
211
212 //Report progress
213 qInfo() << foldCounter << "/" << folds.size() << "folds finished!";
214 }
215
216
217 //Calculate and store overall performance
218 overallPerformance( totalConfusionMatrixValues, aDataPath, "/centralAI_overall_performance_info.csv" );
219
220}

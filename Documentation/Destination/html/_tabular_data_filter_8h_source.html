152 inline bool isEqual( const QVariantList& aFirst, const QVariantList& aSecond )
153 {
154
155 double difference = 0.0;
156
157 for ( int vectorIndex = 0; vectorIndex < aFirst.size(); ++vectorIndex )
158 {
159 difference += std::abs( aFirst.at( vectorIndex ).toDouble() - aSecond.at( vectorIndex ).toDouble() );
160
161 }
162
163 if ( difference / aFirst.size() < DBL_EPSILON )
164 {
165 return true;
166 }
167 else
168 {
169 return false;
170 }
